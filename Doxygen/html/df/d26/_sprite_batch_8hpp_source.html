112  void createRenderBatches(){
113  // This will store all the vertices that we need to upload
114  std::vector <Vertex> vertices;
115  // Resize the buffer to the exact size we need so we can treat
116  // it like an array
117  vertices.resize(_glyphs.size() * 6);
118 
119  if (_glyphs.empty()) {
120  return;
121  }
122 
123  int offset = 0; // current offset
124  int cv = 0; // current vertex
125 
126  //Add the first batch
127  _renderBatches.emplace_back(offset, 6, _glyphs[0]->texture);
128  vertices[cv++] = _glyphs[0]->topLeft;
129  vertices[cv++] = _glyphs[0]->bottomLeft;
130  vertices[cv++] = _glyphs[0]->bottomRight;
131  vertices[cv++] = _glyphs[0]->bottomRight;
132  vertices[cv++] = _glyphs[0]->topRight;
133  vertices[cv++] = _glyphs[0]->topLeft;
134  offset += 6;
135 
136  //Add all the rest of the glyphs
137  for (int cg = 1; cg < _glyphs.size(); cg++) {
138 
139  // Check if this glyph can be part of the current batch
140  if (_glyphs[cg]->texture != _glyphs[cg - 1]->texture) {
141  // Make a new batch
142  _renderBatches.emplace_back(offset, 6, _glyphs[cg]->texture);
143  } else {
144  // If its part of the current batch, just increase numVertices
145  _renderBatches.back().numVertices += 6;
146  }
147  vertices[cv++] = _glyphs[cg]->topLeft;
148  vertices[cv++] = _glyphs[cg]->bottomLeft;
149  vertices[cv++] = _glyphs[cg]->bottomRight;
150  vertices[cv++] = _glyphs[cg]->bottomRight;
151  vertices[cv++] = _glyphs[cg]->topRight;
152  vertices[cv++] = _glyphs[cg]->topLeft;
153  offset += 6;
154  }
155 
156  // Bind our VBO
157  glBindBuffer(GL_ARRAY_BUFFER, _vbo);
158  // Orphan the buffer (for speed)
159  glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(Vertex), nullptr, GL_DYNAMIC_DRAW);
160  // Upload the data
161  glBufferSubData(GL_ARRAY_BUFFER, 0, vertices.size() * sizeof(Vertex), vertices.data());
162 
163  // Unbind the VBO
164  glBindBuffer(GL_ARRAY_BUFFER, 0);
165  }
